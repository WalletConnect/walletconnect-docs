  requiredNamespaces: {
    [namespaceId]: {

Session information proposed and approved by dapp and wallet. [Learn more about sessions](../../../../specs/clients/sign/data-structures#session)

```ts
interface Session {
  topic: string
  relay: {
    protocol: string
  }
  expiry: number
  namespaces: {
    [namespaceId]: {
      methods: string[]
      accounts: string[]
      events: string[]
      chains: string[]
    }
  }
  acknowledged: boolean
  controller: string
  self: {
    publicKey: string
    metadata: {
      name: string
      description: string
      url: string
      icons: string[]
    }
  }
  peer: {
    publicKey: string
    metadata: {
      name: string
      description: string
      url: string
      icons: string[]
    }
  }
}
```
