# Custom Chains

## Custom wagmi chains

<CustomWagmiChainsPartial />

```js
import { w3mProvider } from '@web3modal/ethereum'
import { configureChains } from '@wagmi/core'
import { arbitrum, mainnet, avalanche } from '@wagmi/core/chains'

const chains = [mainnet, avalanche, arbitrum]
const { publicClient } = configureChains(chains, [
  w3mProvider({
    /* ... */
  })
])
```

## Custom default chain

<CustomDefaultChainPartial />

```js
import { optimism } from '@wagmi/core/chains'

const web3modal = new Web3Modal({
  defaultChain: optimism
})
```

## Custom chain providers

<CustomChainProvidersPartial />

```js
import { mainnet } from '@wagmi/core/chains'
import { infuraProvider } from '@wagmi/core/providers/infura'
import { w3mProvider } from '@web3modal/ethereum'

const { chains, publicClient } = configureChains(
  [mainnet],
  [
    w3mProvider({
      /* ... */
    }),
    alchemyProvider({
      /* ... */
    })
  ]
)
```
